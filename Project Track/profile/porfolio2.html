Contact Form Component

Create a form that will send messages to your backend API.

src/components/ContactForm.jsx
import React, { useState } from 'react';

const ContactForm = () => {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    message: ''
  });
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [submitStatus, setSubmitStatus] = useState(null);

  const handleChange = (e) => {
    setFormData({ ...formData, [e.target.name]: e.target.value });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setIsSubmitting(true);
    
    try {
      // We'll implement the API call in Step 6
      setSubmitStatus({ type: 'success', message: 'Message sent successfully!' });
      setFormData({ name: '', email: '', message: '' });
    } catch (error) {
      setSubmitStatus({ type: 'error', message: 'Failed to send message' });
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-6">
      <div>
        <label htmlFor="name" className="block text-gray-700 mb-2">Name</label>
        <input
          type="text"
          id="name"
          name="name"
          value={formData.name}
          onChange={handleChange}
          className="w-full px-4 py-2 border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
          required
        />
      </div>
      
      <div>
        <label htmlFor="email" className="block text-gray-700 mb-2">Email</label>
        <input
          type="email"
          id="email"
          name="email"
          value={formData.email}
          onChange={handleChange}
          className="w-full px-4 py-2 border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
          required
        />
      </div>
      
      <div>
        <label htmlFor="message" className="block text-gray-700 mb-2">Message</label>
        <textarea
          id="message"
          name="message"
          value={formData.message}
          onChange={handleChange}
          rows="5"
          className="w-full px-4 py-2 border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
          required
        ></textarea>
      </div>
      
      <button
        type="submit"
        disabled={isSubmitting}
        className="w-full bg-blue-600 text-white py-3 rounded-lg hover:bg-blue-700 transition disabled:opacity-50"
      >
        {isSubmitting ? 'Sending...' : 'Send Message'}
      </button>
      
      {submitStatus && (
        <div className={`mt-4 p-3 rounded-lg ${
          submitStatus.type === 'success' ? 'bg-green-100 text-green-800' : 'bg-red-100 text-red-800'
        }`}>
          {submitStatus.message}
        </div>
      )}
    </form>
  );
};

export default ContactForm;

Form Features

  • Controlled form inputs with React state
  • Loading state during submission
  • Success/error feedback messages
  • Form validation

Create Express Server

server.js
const express = require('express');
const cors = require('cors');

const app = express();
const port = process.env.PORT || 5000;

// Middleware
app.use(cors());
app.use(express.json());

// Contact form route
app.post('/api/contact', (req, res) => {
  const { name, email, message } = req.body;
  
  // Here you would typically save the message to a database
  // For this example, we'll just log it and send a response
  console.log('New message received:');
  console.log(`Name: ${name}`);
  console.log(`Email: ${email}`);
  console.log(`Message: ${message}`);
  
  // Send success response
  res.status(200).json({
    status: 'success',
    message: 'Thank you for your message!'
  });
});

// Start server
app.listen(port, () => {
  console.log(`Server running on port ${port}`);
});

API Integration

Connect your React frontend to the Node.js backend to handle form submissions.

Update Contact Form

src/components/ContactForm.jsx
// Add this import at the top
import axios from 'axios';

// Inside handleSubmit function
const handleSubmit = async (e) => {
  e.preventDefault();
  setIsSubmitting(true);
  
  try {
    // Send form data to backend API
    const response = await axios.post(
      'http://localhost:5000/api/contact', 
      formData
    );
    
    if (response.data.status === 'success') {
      setSubmitStatus({ type: 'success', message: 'Message sent successfully!' });
      setFormData({ name: '', email: '', message: '' });
    } else {
      setSubmitStatus({ type: 'error', message: 'Failed to send message' });
    }
  } catch (error) {
    setSubmitStatus({ 
      type: 'error', 
      message: error.response?.data?.message || 'Failed to send message'
    });
  } finally {
    setIsSubmitting(false);
  }
};

CORS Configuration

We installed the cors middleware to allow our React app (running on port 3000) to communicate with our backend (running on port 5000).
